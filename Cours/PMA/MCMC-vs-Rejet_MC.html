def Rejet():
    Y = np.random.randn()  # Y ~ N(0,1)
    gY = sps.norm.pdf(Y)
    U = 2*gY*np.random.rand() # U ~ Unif([0,2*g(Y)])
    while U > f(Y):
        Y = np.random.randn()
        U = 2*sps.norm.pdf(Y)*np.random.rand()
    return Y

# Algorithme d'chantillonnage Metropolis-Hastings 

# on fait des pas ~ N(0,sigma^2) autour de l'ancienne valeur x,
# c'est  dire Q(x,y)=g((x-y)/sigma)/sigma o g est la densit d'une N(0,1)

sigma = 3  # on choisit sigma de maniere  avoir environ 25% des pas accepts

def Metropolis_Hastings(Nsteps):
    x = np.random.randn()
    count = 0
    for i in range(Nsteps):
        y = x + sigma*np.random.randn()
        u = np.random.rand()
        if u < f(y)/f(x):
            x = y
            count += 1
    return x, count

def MCMCHist(Nsteps):
    Ech = []
    x = np.random.randn()
    for i in range(Nsteps):
        y = x + sigma*np.random.randn()
        u = np.random.rand()
        if u < f(y)/f(x):
            x = y
        Ech.append(x)
    return Ech

Question 4 : Même chose en plus grande dimension¶

On condidre maintenant la densit sur $\mathbb R^d$ donne par $$ f(x_1,\ldots,x_d)=\prod_{i=1}^d (1+\sin(3x_i))\frac{e^{-x_i^2/2}}{\sqrt{2\pi}} $$ et on cherche rsoudre l'analogue de la question 3, savoir valuer l'intgrale $$ \int_{\mathbb R^d} \|x\|^2 f(x) dx. $$ Sa valeur thorique est donne par $$ \int_{\mathbb R^d} \|x\|^2 f(x) dx = \sum_{i=1}^d \int_{\mathbb R^d}x_i^2 f(x) dx = \sum_{i=1}^d \int_{\mathbb R}x_i^2(1+\sin(3 x_i) dx_i=d. $$

# Methode Rejet : dimension quelconque

def Rejet_D():
    Y = np.random.multivariate_normal(np.zeros(d),np.identity(d))  # Y ~ g = N(0,I_d)
    gY = np.prod(sps.norm.pdf(Y)) 
    U = (2**d)*gY*np.random.rand() # U ~ Unif([0,2^d*g(Y)])
    while U > f_D(Y):
        Y = np.random.multivariate_normal(np.zeros(d),np.identity(d))
        U = (2**d)*gY*np.random.rand()
    return Y


def EstRejet_D(n): 
    L = [sum(Rejet_D()**2) for i in range(n)]
    return(sum(L)/n)


def Precision_RejetMC_D(Nsteps):
    t = time() 
    Est = EstRejet_D(Nsteps)
    Time = time()-t
    Error = abs(Est-d)
    print('Steps =', Nsteps,'\n','Error =',Error,'\n','Time =',Time)

def EstMCMC_D(n): 
    S = 0
    x = np.random.multivariate_normal(np.zeros(d),np.identity(d))
    for i in range(n):
        y = x + sigma*np.random.multivariate_normal(np.zeros(d),np.identity(d))
        u = np.random.rand()
        if u < f_D(y)/f_D(x):
            x = y
        S += sum(x**2)
    return S/n
