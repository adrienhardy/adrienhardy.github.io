# Simulation d'un processus de Poisson de paramtre "lambd"

def Simu_Poisson(t,lambd):
    T = 0
    Z = 0 
    Time_list = [T]
    Ech_list = [Z]
    
    while T < t:
        T += -np.log(npr.rand())/lambd # ~Exp(lambd)
        if T < t:
            Time_list.append(T)
            Z += 1
            Ech_list.append(Z)
        else: 
            Time_list.append(t)
            Ech_list.append(Z)  
            
    return(Time_list,Ech_list)

# Simulation d'une CMTC de gnrateur infinitesimal Q et de mesure initiale mu sur un espace d'tats E 

def Simu_init(mu):
    i = random.choices(range(len(mu)), mu)[0]
    return(E[i])


def Simu_CMTC(t,Q,mu):
    N = len(Q[0]) # taille de E
    T = 0
    Time_list = [T]
    Z = Simu_init(mu)
    Ech_list = [Z]
    lambd = -np.diag(Q) # paramtres des exponentielles entre deux sauts
    P = (Q.T/lambd).T + np.identity(N)  # matrice de transition de la chane de Markov incluse
    
    while T < t:
        i = E.index(Z)
        T += -np.log(npr.rand())/lambd[i]
        if T < t:
            Time_list.append(T)
            j = random.choices(range(N),P[i])[0]
            Z = E[j]
            Ech_list.append(Z)
        else:
            Time_list.append(t)
            Ech_list.append(Z)
            
    return(Time_list,Ech_list)

# Mesure d'quilibre

def Mesure_invariante(Q):
    N = len(Q[0])
    A = np.append(Q.T,[np.ones(N)],axis=0)
    b = np.zeros(N+1) 
    b[-1] = 1
    pi = np.linalg.solve(np.dot(A.T,A),np.dot(A.T,b))
    return(pi)

# Temps moyen pass par tat

def frequences_moyennes(t,Q,mu):
    T,Z = Simu_CMTC(t,Q,mu)
    N = len(Q[0])
    Njumps = len(T)
    frequences = np.zeros(N)
    
    for i in range(Njumps):
        j = E.index(Z[i])
        if i==0:
            frequences[j] += T[1]  # on rappelle que T[0]=0
        elif i<Njumps-1:
            frequences[j] += T[i+1]-T[i]
        else:
            frequences[j] += t-T[-1]
    
    return(frequences/t)

Files d'attente¶

Le generateur $Q$ d'une file d'attente $M/M/K$ de loi d'arrives $\mathcal(\lambda)$ et de loi de dure de traitement $\mathcal E(\mu)$ est donn par $Q_{0,0}=-Q_{0,1}=-\lambda$ puis, pour $n\geq 1$, $$ Q_{x,x-1}=\min(n,K)\mu,\quad Q_{x,x}=-(\lambda+\min(n,K)\mu),\quad Q_{x,x+1}=\lambda $$ et $Q_{x,y}=0$ autrement. Autrement dit, les taux de saut sont $\lambda(n)=\lambda+\min(n,K)\mu$ pour tout $n\in\mathbb N$ et, la matrice de transition de la chane incluse est $P_{0,1}=1$ puis, pour tout $n\geq 1$, $$ P_{n,n-1}=\mu\min(n,K)/(\lambda+\min(n,K)\mu),\qquad P_{n,n+1}=\lambda/(\lambda+\min(n,K)\mu) $$

def MM(K,lambd,mu,t): # inter-arrives ~ Exp(lambd), temps de traitement ~ Exp(mu), K guichets.
    T = 0
    Time_list = [T]
    Z = 0
    Ech_list = [Z]
    
    while T < t:
        T += -np.log(npr.rand())/(lambd+min(Z,K)*mu)
        if T < t:
            Time_list.append(T)
            if npr.rand()<lambd/(lambd+min(Z,K)*mu):
                Z += 1
            else:
                Z -= 1
            Ech_list.append(Z)
        else:
            Time_list.append(t)
            Ech_list.append(Z)
            
    return(Time_list,Ech_list) 


# Taille moyenne empirique de la file 
def moyEst(T,Z):
    Tm = T[:-1] # on supprime T[-1]=t
    Tp = T[1:] # on supprime T[0]=0
    Tdiff = np.array(Tp)-np.array(Tm)
    t = T[-1]
    Zp = Z[:-1] # on supprime Z[-1] (qui est gal  Z[-2])
    return(Tdiff.dot(Zp)/t)
